绘制正弦

clear
clc
x=-2*pi:0.1:2*pi;
y=sin(x);
y1=cos(x);
plot(x,y,'r--')%红色显示曲线
hold on %图形保持
plot(x,y1,'b-')%绘制蓝色余弦曲线
% xlabel({'正弦函数','sin'})%分行显示
xlabel('正弦曲线 余弦曲线')
